add_subdirectory(nevishwutils)

  LIBRARY_NAME
    sbndaq-artdaq_NevisTPCGeneratorBase

   LIBRARIES
    ${LIBRARY_CORE_LIB_LIST}
    sbndaq-artdaq_nevishwutils
)
